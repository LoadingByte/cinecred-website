APT (Debian, Ubuntu, Mint, …)

{{ l10n("linux-repositories.ownRepo").format("APT") }}

  1. {{ l10n("linux-repositories.apt.pgp") }}

    sudo mkdir -m 755 /etc/apt/keyrings/
sudo wget -P /etc/apt/keyrings/ https://cinecred.com/linux-repositories/cinecred.asc

  2. {{ l10n("linux-repositories.apt.repo") }}

    echo 'deb [signed-by=/etc/apt/keyrings/cinecred.asc] https://cinecred.com/debian/ stable main' | sudo tee /etc/apt/sources.list.d/cinecred.list

  3. {{ l10n("linux-repositories.installCommands") }}

    sudo apt update
sudo apt install cinecred

gpgcheck=1
repo_gpgcheck=1
gpgkey=https://cinecred.com/linux-repositories/cinecred.asc

  2. {{ l10n("linux-repositories.installCommands") }}

    sudo yum install cinecred
